def cross_module_correlation_analysis():
    """
    分析不同模块之间的关联关系

    返回:
        correlations: dict - 跨模块关联分析结果
    """
    correlations = {}

    # 品牌维度关联分析
    if (analysis_results['sales_trend'] is not None and
        analysis_results['profit_transfer'] is not None):

        # 提取品牌数据
        try:
            # 从地区销售数据中提取品牌销售数据
            regional_sales = analysis_results['sales_trend']['regional_sales']
            brand_sales = regional_sales.groupby('brand')['total_sales_volume'].sum()

            # 品牌利润数据
            brand_profit = analysis_results['profit_transfer']['brand_profit']

            # 品牌过户数据
            brand_transfer = analysis_results['profit_transfer']['brand_transfer']

            # 合并数据
            brand_analysis = pd.DataFrame({
                'sales_volume': brand_sales,
                'total_profit': brand_profit['total_profit'],
                'profit_margin': brand_profit['avg_margin'],
                'transfer_count': brand_transfer['transfer_count'],
                'avg_transfer_age': brand_transfer['avg_age'],
                'avg_depreciation': brand_transfer['avg_depreciation']
            }).fillna(0)

            # 计算相关性
            brand_correlation = brand_analysis.corr()
            correlations['brand'] = {
                'data': brand_analysis,
                'correlation_matrix': brand_correlation
            }

            print(f"  ✓ 品牌维度关联分析完成 ({len(brand_analysis)} 个品牌)")

            # 找出重要关联
            important_correlations = []
            for i in range(len(brand_correlation.columns)):
                for j in range(i+1, len(brand_correlation.columns)):
                    corr_val = brand_correlation.iloc[i, j]
                    if abs(corr_val) > 0.5:  # 只关注强相关
                        important_correlations.append({
                            'feature1': brand_correlation.columns[i],
                            'feature2': brand_correlation.columns[j],
                            'correlation': corr_val
                        })

            if important_correlations:
                correlations['brand']['important_correlations'] = important_correlations
                print(f"  ✓ 识别 {len(important_correlations)} 个强关联关系")

        except Exception as e:
            print(f"  ⚠ 品牌关联分析失败: {e}")

    # 客户-产品关联分析
    if (analysis_results['customer'] is not None and
        analysis_results['product'] is not None):

        try:
            # 计算客户聚类和购买偏好的关系
            customer_data = analysis_results['customer']['cluster_data']
            product_data = analysis_results['product']['cluster_data']

            # 按时间对齐数据
            customer_monthly = customer_data.groupby('stat_month')['customer_count'].sum()
            product_monthly = product_data.groupby('stat_month')['range_sales'].sum()

            # 合并数据
            customer_product = pd.DataFrame({
                'customer_count': customer_monthly,
                'product_sales': product_monthly
            }).fillna(0)

            # 计算相关性
            cp_correlation = customer_product.corr().iloc[0, 1]
            correlations['customer_product'] = {
                'data': customer_product,
                'correlation': cp_correlation
            }

            print(f"  ✓ 客户-产品关联分析完成 (相关性: {cp_correlation:.3f})")

        except Exception as e:
            print(f"  ⚠ 客户-产品关联分析失败: {e}")

    # 时间维度关联分析
    if analysis_results['sales_trend'] is not None:
        try:
            # Prophet预测结果分析
            forecast_data = analysis_results['sales_trend']['forecast']

            # 提取趋势和季节性成分
            if 'trend' in forecast_data.columns and 'seasonal' in forecast_data.columns:
                trend_seasonal_corr = forecast_data['trend'].corr(forecast_data['seasonal'])
                correlations['time_components'] = {
                    'trend_seasonal_correlation': trend_seasonal_corr
                }
                print(f"  ✓ 时间成分关联分析完成 (趋势-季节性相关性: {trend_seasonal_corr:.3f})")

        except Exception as e:
            print(f"  ⚠ 时间维度关联分析失败: {e}")

    return correlations

correlations = cross_module_correlation_analysis()

def comprehensive_clustering_analysis():
    """
    基于多维度数据进行综合聚类分析

    返回:
        clustering_results: dict - 聚类分析结果
    """
    clustering_results = {}

    # 修复KeyError问题：检查correlations['brand']而不是analysis_results['brand']
    if correlations.get('brand') is not None and len(correlations['brand']['data']) > 1:
        brand_data = correlations['brand']['data']

        # 标准化数据
        scaler = StandardScaler()
        brand_scaled = scaler.fit_transform(brand_data)

        # 确定最佳聚类数（确保K范围有效）
        max_k = min(6, len(brand_data))
        if max_k >= 2:
            silhouette_scores = []
            K_range = range(2, max_k)

            for k in K_range:
                kmeans = KMeans(n_clusters=k, random_state=42, n_init=10)
                labels = kmeans.fit_predict(brand_scaled)
                score = silhouette_score(brand_scaled, labels)
                silhouette_scores.append(score)
                print(f"  • K={k}: 轮廓系数 = {score:.3f}")

            # 选择最佳K值
            best_k = K_range[np.argmax(silhouette_scores)]
            print(f"  ✓ 最佳聚类数: {best_k}")

            # 执行最终聚类
            final_kmeans = KMeans(n_clusters=best_k, random_state=42, n_init=10)
            final_labels = final_kmeans.fit_predict(brand_scaled)

            # 添加聚类标签
            brand_data['cluster'] = final_labels

            clustering_results['brand'] = {
                'data': brand_data,
                'model': final_kmeans,
                'scaler': scaler,
                'best_k': best_k,
                'silhouette_scores': silhouette_scores
            }

            print(f"  ✓ 品牌综合聚类完成 ({best_k} 个聚类)")
            print(f"  ✓ 聚类分布: {pd.Series(final_labels).value_counts().to_dict()}")
        else:
            print(f"  ⚠ 品牌数据量不足，无法进行聚类分析 (仅{len(brand_data)}个品牌)")
    else:
        print(f"  ⚠ 无可用的品牌关联数据，跳过聚类分析")

    return clustering_results

clustering_results = comprehensive_clustering_analysis()

def principal_component_analysis():
    """
    对多维度数据进行主成分分析

    返回:
        pca_results: dict - PCA分析结果
    """
    pca_results = {}

    # 修复KeyError问题：检查correlations['brand']
    if correlations.get('brand') is not None and len(correlations['brand']['data']) > 1:
        try:
            # 准备品牌数据
            brand_data = correlations['brand']['data']

            # 执行PCA
            pca = PCA()
            pca_result = pca.fit_transform(StandardScaler().fit_transform(brand_data))

            # 计算解释的方差比
            explained_variance = pca.explained_variance_ratio_
            cumulative_variance = np.cumsum(explained_variance)

            # 确定需要的主成分数量（解释80%方差）
            n_components = np.argmax(cumulative_variance >= 0.8) + 1

            pca_results['brand'] = {
                'pca': pca,
                'explained_variance': explained_variance,
                'cumulative_variance': cumulative_variance,
                'n_components': n_components,
                'pca_result': pca_result
            }

            print(f"  ✓ PCA分析完成")
            print(f"  ✓ 前{n_components}个主成分解释 {cumulative_variance[n_components-1]:.1%} 方差")

            # 输出主成分负荷
            print(f"\n  前{n_components}个主成分的重要特征负荷:")
            for i in range(min(n_components, 3)):  # 显示前3个主成分
                component_loadings = pd.DataFrame({
                    'feature': brand_data.columns,
                    'loading': pca.components_[i]
                }).sort_values('loading', key=abs, ascending=False)

                print(f"\n  主成分 {i+1} (解释 {explained_variance[i]:.1%} 方差):")
                for idx, row in component_loadings.head(3).iterrows():
                    print(f"    {row['feature']}: {row['loading']:.3f}")

        except Exception as e:
            print(f"  ⚠ PCA分析失败: {e}")
    else:
        print(f"  ⚠ 无可用的品牌数据，跳过PCA分析")

    return pca_results

pca_results = principal_component_analysis()

def generate_comprehensive_insights():

    """

    if clustering_results.get('brand') is not None:
        brand_clusters = clustering_results['brand']['data']

        # 分析每个聚类的特征
        cluster_profiles = {}
        for cluster_id in sorted(brand_clusters['cluster'].unique()):
            cluster_data = brand_clusters[brand_clusters['cluster'] == cluster_id]

            profile = {
                'size': len(cluster_data),
                'avg_sales': cluster_data['sales_volume'].mean(),
                'avg_profit': cluster_data['total_profit'].mean(),
                'avg_margin': cluster_data['profit_margin'].mean(),
                'avg_transfer_rate': cluster_data['transfer_count'].mean(),
                'brands': cluster_data.index.tolist()
            }

            cluster_profiles[f'cluster_{cluster_id}'] = profile

        insights['brand_clusters'] = cluster_profiles

        print(f"  ✓ 识别 {len(cluster_profiles)} 个品牌聚类类型")

        # 识别明星品牌和问题品牌
        if 'profit_margin' in brand_clusters.columns and 'sales_volume' in brand_clusters.columns:
            # 高销售 + 高利润
            star_brands = brand_clusters[
                (brand_clusters['sales_volume'] > brand_clusters['sales_volume'].median()) &
                (brand_clusters['profit_margin'] > brand_clusters['profit_margin'].median())
            ]

            # 低销售 + 低利润
            problem_brands = brand_clusters[
                (brand_clusters['sales_volume'] < brand_clusters['sales_volume'].median()) &
                (brand_clusters['profit_margin'] < brand_clusters['profit_margin'].median())
            ]

            insights['star_brands'] = star_brands.index.tolist()
            insights['problem_brands'] = problem_brands.index.tolist()

            print(f"  ✓ 明星品牌: {len(star_brands)} 个")
            print(f"  ✓ 问题品牌: {len(problem_brands)} 个")

    # 客户价值洞察
    if analysis_results['customer'] is not None:
        clv_data = analysis_results['customer']['clv_data']

        # 高价值客户群体
        top_clv = clv_data.head(3)
        insights['high_value_segments'] = [
            {
                'segment': row['customer_segment'],
                'clv': row['clv'],
                'customer_count': row['total_customers']
            }
            for _, row in top_clv.iterrows()
        ]

        print(f"  ✓ 识别 {len(insights['high_value_segments'])} 个高价值客户群体")

    # 产品趋势洞察
    if analysis_results['product'] is not None:
        importance_data = analysis_results['product']['importance']

        if not importance_data.empty:
            # 最重要的产品特征
            top_features = importance_data.head(3)
            insights['key_product_features'] = [
                {
                    'feature': row['feature'],
                    'importance': row['importance']
                }
                for _, row in top_features.iterrows()
            ]

            print(f"  ✓ 识别 {len(insights['key_product_features'])} 个关键产品特征")

    # 时间趋势洞察
    if analysis_results['sales_trend'] is not None:
        forecast_data = analysis_results['sales_trend']['forecast']

        if 'yhat' in forecast_data.columns:
            # 未来预测
            future_forecast = forecast_data.tail(12)['yhat'].values
            total_forecast = future_forecast.sum()
            avg_forecast = future_forecast.mean()

            insights['sales_forecast'] = {
                'next_12_months_total': total_forecast,
                'next_12_months_average': avg_forecast,
                'growth_trend': 'positive' if future_forecast[-1] > future_forecast[0] else 'negative'
            }

            print(f"  ✓ 未来12个月预测销售额: ¥{total_forecast:,.2f}")

    return insights
